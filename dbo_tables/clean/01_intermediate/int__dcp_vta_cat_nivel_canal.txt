with de_duplicacion_vtacatnivelcanal as (
	select
		*,
		row_number() over(
			partition by clanivelcanal
			order by
				fechaultimamod desc
		) as rownum
	from
		{{ ref('stg__vta_cat_nivel_canal') }}
),
sel_vtacatnivelcanal as (
	select
		_airbyte_raw_id,
		_airbyte_extracted_at,
		_airbyte_meta,
		bajalogica,
		nombrepcmod,
		_ab_cdc_lsn,
		clanivelcanal,
		clausuariomod,
		nomnivelcanal,
		fechaultimamod,
		_ab_cdc_cursor,
		fechabajalogica,
		nomnivelcanaling,
		_ab_cdc_deleted_at,
		_ab_cdc_updated_at,
		_ab_cdc_event_serial_no
		from de_duplicacion_vtacatnivelcanal
	where
		rownum = 1
)
select
	*
from
	sel_vtacatnivelcanal
