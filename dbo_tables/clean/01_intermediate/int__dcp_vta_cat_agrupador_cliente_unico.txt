with de_duplicacion_vtacatagrupadorclienteunico as (
	select
		*,
		row_number() over(
			partition by claagrupadorclienteunico,claagrupadorclienteunicosfid
			order by
				fechaultimamod desc
		) as rownum
	from
		{{ ref('stg__vta_cat_agrupador_cliente_unico') }}
),
sel_vtacatagrupadorclienteunico as (
	select
		_airbyte_raw_id,
		_airbyte_extracted_at,
		_airbyte_meta,
		cucsfid,
		clamarca,
		bajalogica,
		nombrepcmod,
		_ab_cdc_lsn,
		clausuariomod,
		fechaultimamod,
		_ab_cdc_cursor,
		fechabajalogica,
		_ab_cdc_deleted_at,
		_ab_cdc_updated_at,
		_ab_cdc_event_serial_no,
		claagrupadorclienteunico,
		nomagrupadorclienteunico,
		claagrupadorclienteunicosfid
		from de_duplicacion_vtacatagrupadorclienteunico
	where
		rownum = 1
)
select
	*
from
	sel_vtacatagrupadorclienteunico
