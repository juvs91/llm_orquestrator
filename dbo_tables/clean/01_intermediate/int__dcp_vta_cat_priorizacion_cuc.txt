with de_duplicacion_vtacatpriorizacioncuc as (
	select
		*,
		row_number() over(
			partition by clapriorizacion
			order by
				fechaultimamod desc
		) as rownum
	from
		{{ ref('stg__vta_cat_priorizacion_cuc') }}
),
sel_vtacatpriorizacioncuc as (
	select
		_airbyte_raw_id,
		_airbyte_extracted_at,
		_airbyte_meta,
		bajalogica,
		nombrepcmod,
		_ab_cdc_lsn,
		clausuariomod,
		fechaultimamod,
		_ab_cdc_cursor,
		clapriorizacion,
		fechabajalogica,
		nompriorizacion,
		_ab_cdc_deleted_at,
		_ab_cdc_updated_at,
		nomcortopriorizacion,
		_ab_cdc_event_serial_no
		from de_duplicacion_vtacatpriorizacioncuc
	where
		rownum = 1
)
select
	*
from
	sel_vtacatpriorizacioncuc
