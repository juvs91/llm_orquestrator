	select
		*,
		row_number() over(
			partition by clanivelofertanueva
			order by

		) as rownum
	from
		{{ ref('stg__vta_cat_nivel_oferta_nueva') }}

	select

		nombrepcmod,
		_ab_cdc_lsn,

		fechaultimamod,
		_ab_cdc_cursor,
		fechabajalogica,

		clanivelofertanueva,

		de_duplicacion_vtacatnivelofertanueva

		rownum = 1

	*
from
	sel_vtacatnivelofertanueva
